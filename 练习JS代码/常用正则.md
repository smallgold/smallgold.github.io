```javascript
// 非零的正整数
/^[1-9]\d*$/

// 非零的负整数
/^-[1-9]\d*$/

// 正数、负数、和小数
/^(\-|\+)?\d+(\.\d+)?$/

// 有两位小数的正实数
/^[0-9]+(.[0-9]{2})?$/

// 正数、负数、和小数
/^(\-|\+)?\d+(\.\d+)?$/

// 手机号码
/^[1][35789]\d{9}$/ 

// 电话号码
/^(\(\d{3,4}-)|\d{3.4}-)?\d{7,8}$/ 

// 汉字
/^[\u4e00-\u9fa5]{0,}$/

// 英文和数字
/^[A-Za-z0-9]+$/

// 由数字、26个英文字母或者下划线组成的字符串
/^\w+$/

// Email地址
/^\w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/

// 域名
/[a-zA-Z0-9][-a-zA-Z0-9]{0,62}(/.[a-zA-Z0-9][-a-zA-Z0-9]{0,62})+/.?/

// InternetURL
/^(http|https)://([\w-]+\.)+[\w-]+(/[\w-./?%&=]*)?$/

// 1到3个数字,后面跟着任意个 逗号+3个数字,逗号成为可选,而不是必须
/^([1-9]+|[1-9]{1,3}(,[0-9]{3})*)(.[0-9]{1,2})?$/

// 空白行的正则表达式
/\n\s*\r/

// 密码强度正则，最少6位，包括至少1个大写字母，1个小写字母，1个数字，1个特殊字符
/^.*(?=.{6,})(?=.*\d)(?=.*[A-Z])(?=.*[a-z])(?=.*[!@#$%^&*? ]).*$/;

// 车牌号正则
/^[京津沪渝冀豫云辽黑湘皖鲁新苏浙赣鄂桂甘晋蒙陕吉闽贵粤青藏川宁琼使领A-Z]{1}[A-Z]{1}[A-Z0-9]{4}[A-Z0-9挂学警港澳]{1}$/
```
